def to_dot
  lines = @task_graph.tsort.map { |t| serialize_row(t) }
  joined_lines = lines.flatten.map { |l| "  #{l}" }.join("\n").strip

  <<~DOT
    digraph TaskGraph {
      #{joined_lines}
    }
  DOT
end
